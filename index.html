- វិទ្យាសាស្ត្រកុំព្យូទ័រ គឺជាការសិក្សាអំពីកុំព្យូទ័រ និងប្រព័ន្ធកុំព្យូទ័រ។ វារួមបញ្ចូលការរចនា ការអភិវឌ្ឍន៍ និងការវិភាគនៃផ្នែករឹង កម្មវិធី និងក្បួនដោះស្រាយ។ អ្នកវិទ្យាសាស្ត្រកុំព្យូទ័រមានការព្រួយបារម្ភជាមួយនឹងគោលការណ៍គ្រឹះនៃការគណនា និងការអនុវត្តជាក់ស្តែងនៃបច្ចេកវិទ្យាកុំព្យូទ័រ។ ពួកគេស្វែងរកប្រធានបទដូចជា ក្បួនដោះស្រាយ រចនាសម្ព័ន្ធទិន្នន័យ ភាសាសរសេរកម្មវិធី បញ្ញាសិប្បនិម្មិត ស្ថាបត្យកម្មកុំព្យូទ័រ និងច្រើនទៀត។

- បច្ចេកវិទ្យាព័ត៌មាន គឺជាការអនុវត្តជាក់ស្តែងនៃបច្ចេកវិទ្យាកុំព្យូទ័រ។ វាផ្តោតលើការប្រើប្រាស់កុំព្យូទ័រ និងកម្មវិធីដើម្បីគ្រប់គ្រង និងដំណើរការព័ត៌មាន។ អ្នកជំនាញផ្នែកព័ត៌មានវិទ្យាមានទំនួលខុសត្រូវក្នុងការរចនា ការដាក់ពង្រាយ និងថែរក្សាប្រព័ន្ធ IT ដែលគាំទ្រដល់អាជីវកម្ម និងអង្គការនានា។ ពួកគេធ្វើការជាមួយផ្នែករឹង កម្មវិធី បណ្តាញ និងមូលដ្ឋានទិន្នន័យ ដើម្បីធានាថាប្រព័ន្ធទាំងនេះអាចទុកចិត្តបាន ប្រសិទ្ធភាព និងសុវត្ថិភាព។

Software Engineering

វិស្វកម្មកម្មវិធី គឺជាមុខវិជ្ជាដែលពាក់ព័ន្ធនឹងការរចនា បង្កើត និងថែទាំកម្មវិធីកុំព្យូទ័រ។ វាផ្តោតលើការអនុវត្តគោលការណ៍វិស្វកម្ម និងវិធីសាស្រ្តដើម្បីបង្កើតផលិតផលកម្មវិធីដែលមានគុណភាពខ្ពស់។វិស្វករកម្មវិធី ប្រើប្រាស់ចំណេះដឹងរបស់ពួកគេអំពីភាសាសរសេរកម្មវិធី ក្បួនដោះស្រាយ រចនាសម្ព័ន្ធទិន្នន័យ និងដំណើរការអភិវឌ្ឍកម្មវិធីដើម្បីវិភាគតម្រូវការអ្នកប្រើប្រាស់ រចនាដំណោះស្រាយកម្មវិធី សរសេរ និងសាកល្បងកូដ និងធានានូវភាពជឿជាក់ និងដំណើរការនៃប្រព័ន្ធកម្មវិធី។

នៅពេលបញ្ចប់ការសិក្សាពីកម្មវិធីវិស្វកម្មសូហ្វវែរ និស្សិតនឹងមានមូលដ្ឋានរឹងមាំនៅក្នុងគោលការណ៍ និងការអនុវត្តនៃការអភិវឌ្ឍន៍កម្មវិធី។ ពួកគេនឹងអាច៖

  • វដ្តនៃការអភិវឌ្ឍន៍កម្មវិធី៖ ការអភិវឌ្ឍន៍កម្មវិធីទាំងមូល រួមទាំងការប្រមូលផ្តុំតម្រូវការ ការរចនា ការសរសេរកូដ ការធ្វើតេស្ត ការដាក់ឱ្យប្រើប្រាស់ និងការថែទាំ។
  • ភាសា និងបច្ចេកវិទ្យានៃការសរសេរកម្មវិធី៖ ជំនាញក្នុងភាសាសរសេរកម្មវិធីជាច្រើនដូចជា Java, C++, Python, JavaScript ហើយមានចំណេះដឹងអំពីក្របខ័ណ្ឌផ្សេងៗ បណ្ណាល័យ និងឧបករណ៍ដែលទាក់ទងនឹងការអភិវឌ្ឍន៍កម្មវិធី។
  • រចនាសម្ព័ន្ធទិន្នន័យ និងក្បួនដោះស្រាយ៖ សិស្សនឹងមានមូលដ្ឋានគ្រឹះដ៏រឹងមាំនៅក្នុងរចនាសម្ព័ន្ធទិន្នន័យ និងក្បួនដោះស្រាយ ដែលអាចឱ្យពួកគេវិភាគ និងដោះស្រាយបញ្ហាស្មុគស្មាញក្នុងការគណនាប្រកបដោយប្រសិទ្ធភាព។
  • ការរចនាកម្មវិធី និងស្ថាបត្យកម្ម៖ សិស្សនឹងត្រូវបានបំពាក់ដោយចំណេះដឹងអំពីគោលការណ៍រចនាកម្មវិធី គំរូនៃការរចនា និងគំរូស្ថាបត្យកម្ម។
  • ការគ្រប់គ្រងមូលដ្ឋានទិន្នន័យ៖ សិស្សនឹងយល់ពីគោលគំនិត និងគោលការណ៍នៃប្រព័ន្ធគ្រប់គ្រងមូលដ្ឋានទិន្នន័យ រួមទាំងការធ្វើគំរូទិន្នន័យ ភាសាសួរសំណួរ (ឧ. SQL) និងបច្ចេកទេសបង្កើនប្រសិទ្ធភាពមូលដ្ឋានទិន្នន័យ។
  • ការគ្រប់គ្រងគម្រោង៖ សិស្សនឹងមានការយល់ដឹងជាមូលដ្ឋានអំពីគោលការណ៍នៃការគ្រប់គ្រងគម្រោង រួមទាំងការរៀបចំគម្រោង ការរៀបចំកាលវិភាគ ការគ្រប់គ្រងហានិភ័យ និងការសហការជាក្រុម។
  • សុវត្ថិភាពកម្មវិធី៖ សិស្សនឹងដឹងអំពីភាពងាយរងគ្រោះផ្នែកទន់ទូទៅ ការអនុវត្តល្អបំផុតសម្រាប់ការសរសេរកូដសុវត្ថិភាព និងបច្ចេកទេសសម្រាប់ការពារកម្មវិធី និងទិន្នន័យពីការចូលប្រើ ឬការវាយប្រហារដោយគ្មានការអនុញ្ញាត។

បណ្តាញកុំព្យូទ័រគឺជាការអនុវត្តនៃការតភ្ជាប់កុំព្យូទ័រ និងឧបករណ៍ផ្សេងទៀតដើម្បីជួយសម្រួលដល់ការផ្ទេរទិន្នន័យ និងធនធាន។ វាពាក់ព័ន្ធនឹងការរចនា ការអនុវត្ត ការគ្រប់គ្រង និងការថែរក្សាបណ្តាញដែលអនុញ្ញាតឱ្យទំនាក់ទំនង និងការចែករំលែកទិន្នន័យរវាងឧបករណ៍។ បណ្តាញកុំព្យូទ័រអាចជាបណ្តាញក្នុងតំបន់ (LANs) នៅក្នុងតំបន់ភូមិសាស្ត្រតូចមួយ ឬបណ្តាញតំបន់ធំទូលាយ (WANs) ដែលលាតសន្ធឹងលើតំបន់ធំជាង។ ម្យ៉ាងវិញទៀត សន្តិសុខតាមអ៊ីនធឺណិតផ្តោតលើការការពារប្រព័ន្ធកុំព្យូទ័រ បណ្តាញ និងទិន្នន័យពីការចូលប្រើដោយគ្មានការអនុញ្ញាត ការវាយប្រហារ និងការខូចខាត។ វាពាក់ព័ន្ធនឹងការអនុវត្តវិធានការសន្តិសុខ ដើម្បីធានាបាននូវភាពសម្ងាត់ សុចរិតភាព និងលទ្ធភាពទទួលបានព័ត៌មាន។

បន្ទាប់ពីបញ្ចប់ការសិក្សាពីកម្មវិធី Computer Networking and Cyber Security និស្សិតនឹងទទួលបានការយល់ដឹងយ៉ាងទូលំទូលាយអំពីគំនិត ជំនាញ និងឧបករណ៍ផ្សេងៗដែលទាក់ទងនឹងវិស័យនេះ។ នេះគឺជាផ្នែកសំខាន់ៗនៃចំណេះដឹងដែលសិស្សអាចរំពឹងថានឹងទទួលបាន៖

  • ស្ថាបត្យកម្មបណ្តាញ៖ សិស្សនឹងស្គាល់ពីប្រភេទផ្សេងគ្នានៃបណ្តាញ ទំនាក់ទំនងបណ្តាញ ពិធីការ និងធាតុផ្សំហេដ្ឋារចនាសម្ព័ន្ធ។
  • មូលដ្ឋានគ្រឹះសុវត្ថិភាពតាមអ៊ីនធឺណិត៖ សិស្សនឹងយល់ច្បាស់អំពីគោលការណ៍សុវត្ថិភាពតាមអ៊ីនធឺណិត ការគំរាមកំហែង និងភាពងាយរងគ្រោះ។
  • ការរចនាបណ្តាញសុវត្ថិភាព៖ សិស្សនឹងរៀនរចនា និងអនុវត្តបណ្តាញសុវត្ថិភាពដោយប្រើប្រាស់ការអ៊ិនគ្រីប ជញ្ជាំងភ្លើង ប្រព័ន្ធរកឃើញការឈ្លានពាន (IDS) និងវិធានការសុវត្ថិភាពផ្សេងទៀត។.
  • ការឆ្លើយតបឧប្បត្តិហេតុ៖ សិស្សនឹងទទួលបានចំណេះដឹងអំពីនីតិវិធីឆ្លើយតបឧប្បត្តិហេតុ និងការអនុវត្តល្អបំផុត។
  • Ethical Hacking៖ សិស្សនឹងទទួលបានជំនាញ និងចំណេះដឹងទាក់ទងនឹងការលួចចូលប្រកបដោយក្រមសីលធម៌ និងការធ្វើតេស្តការជ្រៀតចូល។ ពួកគេនឹងយល់ពីវិធីសាស្រ្ត ឧបករណ៍ និងបច្ចេកទេសដែលប្រើដើម្បីកំណត់ និងទាញយកភាពងាយរងគ្រោះនៅក្នុងប្រព័ន្ធកុំព្យូទ័រ និងបណ្តាញ។
  • ការគ្រប់គ្រងសុវត្ថិភាព៖ សិស្សនឹងយល់អំពីគោលគំនិតនៃការគ្រប់គ្រងសន្តិសុខ និងការគ្រប់គ្រងហានិភ័យ។ ពួកគេនឹងស៊ាំជាមួយក្របខ័ណ្ឌសុវត្ថិភាព (ឧ. NIST, ISO 27001) និងយល់ពីដំណើរការដែលពាក់ព័ន្ធនឹងការវាយតម្លៃ កាត់បន្ថយ និងការត្រួតពិនិត្យហានិភ័យ។

ឱកាសការងារក្រោយឧត្តមសិក្សា ផ្នែកបណ្តាញកុំព្យូទ័រ និងសន្តិសុខតាមអ៊ីនធឺណិត៖

  • Network Security Analyst(អ្នកវិភាគសុវត្ថិភាពបណ្តាញ)
  • Cybersecurity Consultant(ទីប្រឹក្សាសន្តិសុខតាមអ៊ីនធឺណិត)
  • Information Security Manager(អ្នកគ្រប់គ្រងសន្តិសុខព័ត៌មាន)
  • Network Engineer(វិស្វករបណ្តាញ)
  • Security Software Developer(អ្នកបង្កើតកម្មវិធីសុវត្ថិភាព)
  • Incident Response Specialist(អ្នកឯកទេសឆ្លើយតបឧប្បត្តិហេតុ)
  • Cybersecurity Policy Analyst(អ្នកវិភាគគោលនយោបាយសន្តិសុខអ៊ីនធឺណិត)
  • Cyber Threat Intelligence Analyst(អ្នកវិភាគការគំរាមកំហែងតាមអ៊ីនធឺណិត)

Computer Multimedia Design

Computer Multimedia Design សំដៅលើការបង្កើត និងរចនាមាតិកាប្រព័ន្ធផ្សព្វផ្សាយឌីជីថល ដែលរួមបញ្ចូលគ្នានូវទម្រង់ផ្សេងៗនៃមេឌៀ ដូចជា អត្ថបទ រូបភាព ក្រាហ្វិក អូឌីយ៉ូ វីដេអូ និងគំនូរជីវចល ដើម្បីផ្តល់នូវបទពិសោធន៍អន្តរកម្ម និងការចូលរួម។ អ្នករចនាពហុព័ត៌មានកុំព្យូទ័រប្រើប្រាស់ការរួមបញ្ចូលគ្នានៃជំនាញបច្ចេកទេស និងការច្នៃប្រឌិត ដើម្បីអភិវឌ្ឍបទបង្ហាញពហុព័ត៌មាន គេហទំព័រ កម្មវិធីទូរស័ព្ទ ហ្គេមវីដេអូ បញ្ជរអន្តរកម្ម សម្ភារៈសិក្សាតាមអេឡិចត្រូនិក និងមាតិកាឌីជីថលផ្សេងទៀត។

ចំណេះដឹង និងជំនាញសំខាន់ៗដែលទទួលបានដោយនិស្សិតបញ្ចប់ការសិក្សារួមមាន៖

  • គោលការណ៍រចនាពហុមេឌៀ៖ ការយល់ដឹងហ្មត់ចត់អំពីគោលការណ៍រចនារូបភាព ទ្រឹស្ដីពណ៌ អក្សរវាយអក្សរ ប្លង់ និងបច្ចេកទេសនិទានរឿងសម្រាប់ទំនាក់ទំនងពហុព័ត៌មានប្រកបដោយប្រសិទ្ធភាព។
  • ជំនាញកម្មវិធីពហុព័ត៌មាន៖ ស្គាល់ឧបករណ៍កម្មវិធីស្តង់ដារឧស្សាហកម្មដូចជា Adobe Creative Suite (Photoshop, Illustrator, After Effects) និងកម្មវិធីឯកទេសផ្សេងទៀតសម្រាប់បង្កើត និងរៀបចំមាតិកាពហុព័ត៌មាន។
  • ជំនាញរចនាអន្តរកម្ម និងគំរូគំរូ៖ ការយល់ដឹងអំពីគោលការណ៍រចនាចំណុចប្រទាក់អ្នកប្រើ (UI) ការពិចារណាលើការរចនាបទពិសោធន៍អ្នកប្រើប្រាស់ (UX) និងឧបករណ៍គំរូដើមដើម្បីបង្កើតបទពិសោធន៍អន្តរកម្មដែលងាយស្រួលប្រើ និងងាយស្រួលប្រើ។
  • ការផលិតគំនូរជីវចល និងវីដេអូ៖ ការយល់ដឹងអំពីបច្ចេកទេសគំនូរជីវចល គោលការណ៍កែសម្រួលវីដេអូ និងកម្មវិធីក្រាហ្វិកចលនា ដើម្បីបង្កើតខ្លឹមសារពហុព័ត៌មានដែលទាក់ទាញភ្នែក និងទាក់ទាញ។
  • ការរចនា និងការរួមបញ្ចូលគ្នា៖ ស្គាល់ជាមួយកម្មវិធីកែសំឡេង បែបផែនសំឡេង និងសមាសភាពតន្ត្រី ដើម្បីបង្កើនការនិទានរឿង និងបទពិសោធន៍អ្នកប្រើប្រាស់នៃគម្រោងពហុព័ត៌មាន។
  • ការរចនាក្រាហ្វិក និងរូបភាព៖ សិស្សនឹងមានជំនាញផ្នែកកម្មវិធី និងបច្ចេកទេសរចនាក្រាហ្វិក ដើម្បីបង្កើតរូបភាព ឡូហ្គោ រូបតំណាង និងធាតុដែលមើលឃើញផ្សេងទៀត។

  • Senior Multimedia Designer(អ្នករចនាពហុព័ត៌មានជាន់ខ្ពស់)

  • Motion Graphics Designer(អ្នករចនាក្រាហ្វិកចលនា)
  • Interactive Media Developer (អ្នកអភិវឌ្ឍន៍ប្រព័ន្ធផ្សព្វផ្សាយអន្តរកម្ម )
